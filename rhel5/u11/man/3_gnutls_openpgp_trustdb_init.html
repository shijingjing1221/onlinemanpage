NAME gnutls_openpgp_trustdb_init - This function initializes a gnutls_openpgp_trustdb_t structure

SYNOPSIS #include <gnutls/openpgp.h>

int gnutls_openpgp_trustdb_init(gnutls_openpgp_trustdb_t * trustdb);

ARGUMENTS gnutls_openpgp_trustdb_t * trustdb The structure to be initialized

DESCRIPTION This function will initialize an OpenPGP trustdb structure.

Returns 0 on success.

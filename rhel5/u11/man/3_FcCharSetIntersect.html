NAME FcCharSetIntersect - Intersect charsets

SYNOPSIS #include <fontconfig.h>

FcCharSet * FcCharSetIntersect (const FcCharSet *a, const FcCharSet *b);

DESCRIPTION Returns a set including only those chars found in both a and b.

VERSION Fontconfig version 2.4.1

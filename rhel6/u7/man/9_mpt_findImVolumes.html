NAME mpt_findImVolumes - Identify IDs of hidden disks and RAID Volumes

SYNOPSIS int mpt_findImVolumes(MPT_ADAPTER * ioc);

ARGUMENTS ioc Pointer to a Adapter Strucutre

RETURN 0 on success -EFAULT if read of config page header fails or data pointer not NULL -ENOMEM if pci_alloc failed

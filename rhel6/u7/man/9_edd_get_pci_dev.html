NAME edd_get_pci_dev - finds pci_dev that matches edev

SYNOPSIS struct pci_dev * edd_get_pci_dev(struct edd_device * edev);

ARGUMENTS edev edd_device

DESCRIPTION Returns pci_dev if found, or NULL

NAME gnutls_session_is_resumed - check whether this session is a resumed one

SYNOPSIS #include <gnutls/gnutls.h>

int gnutls_session_is_resumed(gnutls_session_t session);

ARGUMENTS gnutls_session_t session is a gnutls_session_t structure.

DESCRIPTION Check whether session is resumed or not.

RETURNS non zero if this session is resumed, or a zero if this is a new ses- sion.

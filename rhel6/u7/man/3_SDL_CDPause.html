NAME SDL_CDPause - Pauses a CDROM

SYNOPSIS #include "SDL.h"

int SDL_CDPause(SDL_CD *cdrom);

DESCRIPTION Pauses play on the given cdrom.

RETURN VALUE Returns 0 on success, or -1 on an error.

NAME SDL_CDStop - Stops a CDROM

SYNOPSIS #include "SDL.h"

int SDL_CDStop(SDL_CD *cdrom);

DESCRIPTION Stops play on the given cdrom.

RETURN VALUE Returns 0 on success, or -1 on an error.

NAME fb_mode_is_equal - compare 2 videomodes

SYNOPSIS int fb_mode_is_equal(const struct fb_videomode * mode1, const struct fb_videomode * mode2);

ARGUMENTS mode1 first videomode

mode2 second videomode

RETURNS 1 if equal, 0 if not

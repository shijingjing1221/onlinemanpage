NAME mptspi_is_raid - Determines whether target is belonging to volume

SYNOPSIS int mptspi_is_raid(struct _MPT_SCSI_HOST * hd, u32 id);

ARGUMENTS hd Pointer to a SCSI HOST structure

id target device id

RETURN non-zero = true zero = false

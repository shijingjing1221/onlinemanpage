NAME

ldns_rr_list_new, ldns_rr_list_free

SYNOPSIS

#include <stdint.h>
#include <stdbool.h>

#include <ldns/ldns.h>

ldns_rr_list* ldns_rr_list_new();

void ldns_rr_list_free(ldns_rr_list *rr_list);

DESCRIPTION

ldns_rr_list_new() creates a new rr_list structure.

Returns a new rr_list structure

ldns_rr_list_free() frees an rr_list structure.

rr_list: the list to free

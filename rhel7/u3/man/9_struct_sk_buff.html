NAME

struct_sk_buff − socket buffer

SYNOPSIS

struct sk_buff {
struct sk_buff * next;
struct sk_buff * prev;
#ifdef __GENKSYMS__
ktime_t tstamp;
#else
union {unnamed_union};
#endif
__u16 inner_transport_header;
__u16 inner_network_header;
__u16 inner_mac_header;
__u16 transport_header;
__u16 network_header;
__u16 mac_header;
sk_buff_data_t tail;
sk_buff_data_t end;
unsigned int truesize;
atomic_t users;
};
